Vous devez combiner deux méthodes en utilisant la boucle for et la condition if. Dans la condition if, nous devons vérifier si la chaîne est vide ou non, en utilisant len(). 

text = "Hello, world!"
# Use for loop to iterate the splitted text
# ("Hello," and "world!")
for i in text.split():  
    # Check if the strings are not empty                    
    if len(i.strip()) > 0:
        # Print "OK" if all good

def non_empty_lines(text: str) -> int:

    t = 0
    for i in text.split('\n'):
        if len(i.strip()) > 0:
            t += 1
    return t

Cette option vous permet de réduire votre code et de le rendre plus léger en supprimant les variables redondantes du type t = 0. 

def non_empty_lines(text: str) -> int:
    # Here we are not using a row counter, but a list of non-empty rows.
    # And as a result, we return its length using len()
    return len([x for x in text.split("\n") if x.strip()])
